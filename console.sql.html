93   -- Trigger per garantire che non due biglietti con lo stesso posto sullo stesso volo
94   CREATE TRIGGER check_unique_seat
95   BEFORE INSERT ON biglietto
96   FOR EACH ROW
97   BEGIN
98       DECLARE conteggio_posti INT;
99       SELECT COUNT(*)
100      INTO conteggio_posti
101      FROM biglietto
102      WHERE id_volo = NEW.id_volo AND posto = NEW.posto;
103      IF conteggio_posti > 0 THEN
104          SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Il posto è già stato prenotato per questo volo';
105      END IF;
106  END;

109  -- Trigger per garantire che il numero di biglietti non superi il numero di posti disponibili
110  CREATE TRIGGER disponibilita_posti
111  BEFORE INSERT ON biglietto
112  FOR EACH ROW
113  BEGIN
114      DECLARE conteggio_biglietti INT;
115      DECLARE capienza_posti INT;
116      SELECT COUNT(*)
117      INTO conteggio_biglietti
118      FROM biglietto
119      WHERE id_volo = NEW.id_volo;
120  
121      SELECT numeri_posti
122      INTO capienza_posti
123      FROM volo
124      WHERE id_volo = NEW.id_volo;
125  
126      IF conteggio_biglietti >= capienza_posti THEN
127          SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Il numero dei biglietti venduti supera quello dei posti disponibili';
128      END IF;
129  END;

132  -- Trigger per garantire che la data di check-in sia nel futuro e quella di check-out sia successiva a quella del check-in
133  CREATE TRIGGER controllo_prenotazioni_hotel
134  BEFORE INSERT ON prenotazione_hotel
135  FOR EACH ROW
136  BEGIN
137     IF NEW.data_check_in <= CURRENT_DATE THEN
138         SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'La data di check in deve essere nel futuro';
139     END IF;
140  
141     IF NEW.data_check_out <= NEW.data_check_in THEN
142         SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'La data di check out deve essere successiva a quella di check in';
143     END IF;
144  END;

147  -- Trigger per garantire che la data di ritiro dell'auto sia nel futuro e la data di restituzione sia successiva o uguale alla data di ritiro
148  CREATE TRIGGER controllo_noleggi
149  BEFORE INSERT ON macchine
150  FOR EACH ROW
151  BEGIN
152      IF NEW.ritiro <= CURRENT_DATE THEN
153          SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'La data di ritiro deve essere nel futuro';
154      END IF;
155  
156      IF NEW.restituzione < NEW.ritiro THEN
157          SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'La data di restituzione deve essere successiva o uguale alla data di ritiro';
158      END IF;
159  END;

169      END IF;
